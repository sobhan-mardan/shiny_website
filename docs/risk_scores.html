server <- function(input, output, session) {

  cha_data <- reactive({

    age_points <- if (input$age >= 75) {
      2
    } else if (input$age >= 65) {
      1
    } else {
      0
    }

    sex_points <- ifelse(input$sex == "Female", 1, 0)
    chf_points <- ifelse(input$chf, 1, 0)
    htn_points <- ifelse(input$htn_history, 1, 0)
    diabetes_points <- ifelse(input$diabetes, 1, 0)
    stroke_points <- ifelse(input$stroke_tia, 2, 0)
    vascular_points <- ifelse(input$vascular, 1, 0)

    total_score <-
      chf_points +
      htn_points +
      age_points +
      diabetes_points +
      stroke_points +
      vascular_points +
      sex_points

    total_score <- min(total_score, 9)

    annual_risk <- stroke_risk_lookup$annual_risk[
      stroke_risk_lookup$score == total_score
    ]

    category <- if (input$sex == "Male" && total_score == 0) {
      "Low"
    } else if (input$sex == "Female" && total_score == 1) {
      "Low"
    } else if (input$sex == "Male" && total_score == 1) {
      "Intermediate"
    } else if (input$sex == "Female" && total_score == 2) {
      "Intermediate"
    } else {
      "High"
    }

    interpretation <- if (category == "Low") {
      "Low estimated thromboembolic risk. Anticoagulation is usually not recommended solely based on CHA₂DS₂-VASc."
    } else if (category == "Intermediate") {
      "Intermediate estimated thromboembolic risk. Anticoagulation may be considered using shared decision-making and additional risk modifiers."
    } else {
      "High estimated thromboembolic risk. Oral anticoagulation is generally recommended unless contraindicated."
    }

    list(
      total_score = total_score,
      annual_risk = annual_risk,
      category = category,
      interpretation = interpretation,
      age_points = age_points,
      sex_points = sex_points,
      chf_points = chf_points,
      htn_points = htn_points,
      diabetes_points = diabetes_points,
      stroke_points = stroke_points,
      vascular_points = vascular_points
    )
  })

  hasbled_data <- reactive({

    hypertension_points <- ifelse(input$sbp > 160, 1, 0)
    renal_points <- ifelse(input$renal, 1, 0)
    liver_points <- ifelse(input$liver, 1, 0)
    stroke_points <- ifelse(input$stroke_tia, 1, 0)
    bleeding_points <- ifelse(input$bleeding, 1, 0)
    labile_inr_points <- ifelse(isTRUE(input$vka) && isTRUE(input$labile_inr), 1, 0)
    elderly_points <- ifelse(input$age > 65, 1, 0)
    drugs_points <- ifelse(input$drugs, 1, 0)
    alcohol_points <- ifelse(input$alcohol, 1, 0)

    total_score <-
      hypertension_points +
      renal_points +
      liver_points +
      stroke_points +
      bleeding_points +
      labile_inr_points +
      elderly_points +
      drugs_points +
      alcohol_points

    risk_flag <- ifelse(total_score >= 3, "Elevated", "Not elevated")

    interpretation <- if (total_score >= 3) {
      paste(
        "HAS-BLED score is 3 or higher, indicating elevated bleeding risk.",
        "This should trigger closer follow-up and review of modifiable bleeding-risk factors.",
        "The score should not be used as the only reason to withhold oral anticoagulation."
      )
    } else {
      paste(
        "HAS-BLED score is below 3.",
        "Bleeding risk is not flagged as elevated by this score, but clinical judgment is still required.",
        "Review modifiable risk factors and reassess if clinical status changes."
      )
    }

    modifiable_items <- data.frame(
      Factor = c(
        "Uncontrolled hypertension",
        "Abnormal renal function",
        "Abnormal liver function",
        "Labile INR if receiving VKA",
        "Concomitant antiplatelet therapy or NSAID use",
        "Excess alcohol use"
      ),
      Present = c(
        hypertension_points == 1,
        renal_points == 1,
        liver_points == 1,
        labile_inr_points == 1,
        drugs_points == 1,
        alcohol_points == 1
      ),
      Suggested_review = c(
        "Optimize blood pressure control.",
        "Review renal function, dose appropriateness, and reversible contributors.",
        "Review liver disease severity and reversible contributors.",
        "Improve INR monitoring, adherence, interacting drugs, diet consistency, or consider DOAC suitability when appropriate.",
        "Review need for antiplatelet/NSAID therapy and deprescribe where clinically appropriate.",
        "Counsel on alcohol reduction and reassess bleeding risk."
      )
    )

    list(
      total_score = total_score,
      risk_flag = risk_flag,
      interpretation = interpretation,
      hypertension_points = hypertension_points,
      renal_points = renal_points,
      liver_points = liver_points,
      stroke_points = stroke_points,
      bleeding_points = bleeding_points,
      labile_inr_points = labile_inr_points,
      elderly_points = elderly_points,
      drugs_points = drugs_points,
      alcohol_points = alcohol_points,
      modifiable_items = modifiable_items
    )
  })

  output$cha_score_text <- renderText({
    cha_data()$total_score
  })

  output$cha_category_text <- renderText({
    cha_data()$category
  })

  output$hasbled_score_text <- renderText({
    hasbled_data()$total_score
  })

  output$hasbled_flag_text <- renderText({
    hasbled_data()$risk_flag
  })

  output$combined_interpretation_text <- renderText({
    paste(
      paste0("CHA₂DS₂-VASc: ", cha_data()$total_score, " (", cha_data()$category, " stroke-risk category)."),
      cha_data()$interpretation,
      "",
      paste0("HAS-BLED: ", hasbled_data()$total_score, " (", hasbled_data()$risk_flag, " bleeding-risk flag)."),
      hasbled_data()$interpretation,
      sep = "\n"
    )
  })

  output$review_text <- renderText({
    modifiable <- hasbled_data()$modifiable_items
    present_factors <- modifiable$Factor[modifiable$Present]

    if (length(present_factors) == 0) {
      "No modifiable HAS-BLED factors are currently selected. Continue routine clinical review, medication reconciliation, renal/liver function monitoring when relevant, and reassessment over time."
    } else {
      paste(
        "Selected modifiable bleeding-risk factors:",
        paste0("- ", present_factors, collapse = "\n"),
        sep = "\n"
      )
    }
  })

  output$summary_table <- renderTable({
    data.frame(
      Score = c("CHA₂DS₂-VASc", "HAS-BLED"),
      Purpose = c("Stroke/systemic embolism risk stratification", "Bleeding-risk factor identification"),
      Total = c(cha_data()$total_score, hasbled_data()$total_score),
      Interpretation = c(
        cha_data()$category,
        hasbled_data()$risk_flag
      ),
      Important_note = c(
        "Female sex is included as a point in the score but should be interpreted in clinical context.",
        "A high score should trigger risk-factor modification, not automatic OAC withholding."
      )
    )
  })

  output$cha_component_table <- renderTable({

    cd <- cha_data()

    data.frame(
      Component = c(
        "Congestive heart failure / LV dysfunction",
        "Hypertension",
        "Age",
        "Diabetes mellitus",
        "Prior stroke/TIA/thromboembolism",
        "Vascular disease",
        "Sex category"
      ),
      Value = c(
        ifelse(input$chf, "Yes", "No"),
        ifelse(input$htn_history, "Yes", "No"),
        paste0(input$age, " years"),
        ifelse(input$diabetes, "Yes", "No"),
        ifelse(input$stroke_tia, "Yes", "No"),
        ifelse(input$vascular, "Yes", "No"),
        input$sex
      ),
      Points = c(
        cd$chf_points,
        cd$htn_points,
        cd$age_points,
        cd$diabetes_points,
        cd$stroke_points,
        cd$vascular_points,
        cd$sex_points
      )
    )
  })

  output$hasbled_component_table <- renderTable({

    hd <- hasbled_data()

    data.frame(
      Component = c(
        "Hypertension",
        "Abnormal renal function",
        "Abnormal liver function",
        "Prior stroke",
        "Bleeding history or predisposition",
        "Labile INR",
        "Elderly",
        "Drugs predisposing to bleeding",
        "Alcohol use"
      ),
      Definition = c(
        "Systolic blood pressure >160 mmHg",
        "Dialysis, transplant, or creatinine >200 µmol/L",
        "Cirrhosis or significant biochemical liver abnormality",
        "History of stroke",
        "Prior major bleeding or bleeding predisposition",
        "Unstable/high INR or time in therapeutic range <60%; only counted for VKA therapy",
        "Age >65 years",
        "Concomitant antiplatelet therapy or NSAID use",
        "Excess alcohol use, approximately ≥8 drinks/week"
      ),
      Value = c(
        paste0(input$sbp, " mmHg"),
        ifelse(input$renal, "Yes", "No"),
        ifelse(input$liver, "Yes", "No"),
        ifelse(input$stroke_tia, "Yes", "No"),
        ifelse(input$bleeding, "Yes", "No"),
        ifelse(input$vka, ifelse(isTRUE(input$labile_inr), "Yes", "No"), "Not applicable"),
        paste0(input$age, " years"),
        ifelse(input$drugs, "Yes", "No"),
        ifelse(input$alcohol, "Yes", "No")
      ),
      Points = c(
        hd$hypertension_points,
        hd$renal_points,
        hd$liver_points,
        hd$stroke_points,
        hd$bleeding_points,
        hd$labile_inr_points,
        hd$elderly_points,
        hd$drugs_points,
        hd$alcohol_points
      )
    )
  })

  output$modifiable_table <- renderTable({
    modifiable <- hasbled_data()$modifiable_items
    modifiable$Present <- ifelse(modifiable$Present, "Yes", "No")
    modifiable
  })

  output$cha_risk_plot <- renderPlot({

    current_score <- cha_data()$total_score

    bar_heights <- stroke_risk_lookup$annual_risk
    names(bar_heights) <- stroke_risk_lookup$score

    bar_positions <- barplot(
      height = bar_heights,
      names.arg = stroke_risk_lookup$score,
      xlab = "CHA₂DS₂-VASc score",
      ylab = "Estimated annual stroke risk (%)",
      main = "Example annual stroke risk by CHA₂DS₂-VASc score",
      ylim = c(0, max(bar_heights) + 2)
    )

    points(
      x = bar_positions[current_score + 1],
      y = stroke_risk_lookup$annual_risk[stroke_risk_lookup$score == current_score],
      pch = 19,
      cex = 2
    )

    text(
      x = bar_positions[current_score + 1],
      y = stroke_risk_lookup$annual_risk[stroke_risk_lookup$score == current_score] + 0.8,
      labels = "Selected",
      cex = 0.9
    )
  })

  output$hasbled_component_plot <- renderPlot({

    hd <- hasbled_data()

    component_points <- c(
      "Hypertension" = hd$hypertension_points,
      "Renal" = hd$renal_points,
      "Liver" = hd$liver_points,
      "Stroke" = hd$stroke_points,
      "Bleeding" = hd$bleeding_points,
      "Labile INR" = hd$labile_inr_points,
      "Elderly" = hd$elderly_points,
      "Drugs" = hd$drugs_points,
      "Alcohol" = hd$alcohol_points
    )

    barplot(
      height = component_points,
      horiz = TRUE,
      xlim = c(0, 1.2),
      xlab = "Points",
      main = "HAS-BLED component contribution",
      las = 1
    )
  })
}

shinyApp(ui = ui, server = server)
